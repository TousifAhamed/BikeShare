df=pd.DataFrame({"A":[100,110,150,200,400,100]})
b=[]
c=[]
d=[]
for i in range(len(df)):
    x=int(df.iloc[i].astype(int))-int(df.iloc[i-1].astype(int))
    p=(x/int(df.iloc[i-1].astype(int)))*100
    if x>0:
        b.append(str(int(p))+" % UP")
    elif x<0:
        b.append(str(abs(int(p)))+" % DOWN")
    else:
        b.append(str(int(p))+" %")

    if abs(x)>=25:
        c.append("Alert")
        d.append(int(df.iloc[i].astype(int)))
    else:
        c.append("Chill")
        d.append(0)


df["B"]=b
df["C"]=c
df['D']=d
